% short division visualization
% form https://tex.stackexchange.com/questions/210648/division-with-remainders-displayed-on-side
% other styles: https://tex.stackexchange.com/questions/254633/how-to-type-short-division
\usepackage{stackengine}
\def\rlwd{.5pt} \def\rlht{2.2ex} \def\rldp{.5ex}
\def\mydiv#1{~%
  \rule[-\rldp]{\rlwd}{\rlht}%
  \setbox0=\hbox{~#1}%
  \stackunder[\dimexpr\rldp-\rlwd]{~#1}{\rule{\wd0}{\rlwd}}%
}
